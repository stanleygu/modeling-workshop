# Closed system
cell = '''
    S1 -> S2; k1 * S1 - k2 * S2;
    
    S1 = 10; S2 = 0;
    k1 = 1; k2 = 0.5;
'''
rr = te.loadAntimonyModel(cell)
result = rr.simulate(0, 6, 100)
p = te.plotWithLegend(rr, result)

    S2 -> $X1; k2*S2;
    
    S1= 0.5;
    S2 = 0.333;
    v0 = 1;
    k1 = 2; k2 = 3;
    
    at (time > 40): S1 = 0.6; # Positive perturbation
    at (time > 20): S1 = 0.4; # Negative perturbation
